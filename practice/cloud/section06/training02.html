    # vim multi.yaml
----------------------------------------------------------------------
apiVersion: v1
kind: Pod
metadata:
  creationTimestamp: null
  labels:
    run: hello
  name: hello
spec:
  containers:
  - args:
    - /bin/sh
    - -c
    - echo Hello, Docker & Kubernetes !!;sleep 2400
    image: busybox
    name: hello
    resources: {}
  - args:
    - /bin/sh
    - -c
    - echo Hello, Docker & Kubernetes !!;sleep 2400
