kindでKubernetesクラスタの構築

  1. 「multinode.yaml」の作成 
    # vim multinode.yaml
kind: Cluster
apiVersion: kind.x-k8s.io/v1alpha4
# 1 control plane node and 3 workers
nodes:
# the control plane node config
- role: control-plane
# the three workers
- role: worker
- role: worker
- role: worker
